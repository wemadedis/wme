172 void Renderer::RecordRenderPass()
173 {
174  for (uint32_t bufferIndex = 0; bufferIndex < _commandBufferManager->GetCommandBufferCount(); bufferIndex++)
175  {
176  VkCommandBuffer cmdBuffer = _commandBufferManager->GetCommandBuffer(bufferIndex);
177  _renderPass->BeginRenderPass(cmdBuffer, _swapChain->GetFramebuffers()[bufferIndex], _globalUniform.ClearColor);
178 
179  vkCmdBindPipeline(cmdBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, _pipeline->GetHandle());
180 
181  for (unsigned int meshIndex = 0; meshIndex < _meshInstances.size(); meshIndex++)
182  {
183  MeshInfo *mesh = &_meshes[_meshInstances[meshIndex].mesh];
184  VkBuffer vertexBuffers[] = {mesh->vertexBuffer.buffer};
185  VkDeviceSize offsets[] = {0};
186 
187  vkCmdBindVertexBuffers(cmdBuffer, 0, 1, vertexBuffers, offsets);
188 
189  vkCmdBindIndexBuffer(cmdBuffer, mesh->indexBuffer.buffer, 0, VK_INDEX_TYPE_UINT16);
190 
191  vkCmdBindDescriptorSets(cmdBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, _pipeline->GetLayout(), 0, 1, &_descriptorManager->GetDescriptorSets()[meshIndex], 0, nullptr);
192 
193  vkCmdDrawIndexed(cmdBuffer, static_cast<uint32_t>(mesh->IndexCount), 1, 0, 0, 0);
194  }
195 
196  if(_lineModule != nullptr)
197  {
198  auto lines = _lineModule->GetLineData();
199  if(_lineBuffer.buffer == VK_NULL_HANDLE)
200  {
201  _deviceMemoryManager->CreateBuffer(VK_BUFFER_USAGE_VERTEX_BUFFER_BIT, MemProps::HOST , sizeof(Line)*1000, _lineBuffer);
202  }
203  //Requires at least one mesh instance in order to draw the lines, as the drawing needs a descriptor set for the view & projection matrices.
204  if(lines.size() > 0 && _meshInstances.size() > 0)
205  {
206  _deviceMemoryManager->CopyDataToBuffer(_lineBuffer, lines.data(), sizeof(Line)*lines.size());
207 
208  vkCmdBindPipeline(cmdBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, _linePipeline->GetHandle());
209  VkDeviceSize offsets[] = {0};
210  vkCmdBindDescriptorSets(cmdBuffer, VK_PIPELINE_BIND_POINT_GRAPHICS, _linePipeline->GetLayout(), 0, 1, &_descriptorManager->GetDescriptorSets()[0], 0, nullptr);
211  vkCmdBindVertexBuffers(cmdBuffer, 0, 1, &_lineBuffer.buffer, offsets);
212  vkCmdDraw(cmdBuffer, (uint32_t)lines.size()*2, 1, 0, 0);
213  _lineModule->ClearData();
214  }
215  }
216  _renderPass->NextSubpass(cmdBuffer, VK_SUBPASS_CONTENTS_INLINE);
217  if (_guiModule != nullptr)
218  {
219  _guiModule->Draw(cmdBuffer, _frameWidth, _frameHeight);
220  }
221  _renderPass->EndRenderPass(cmdBuffer);
222  }
