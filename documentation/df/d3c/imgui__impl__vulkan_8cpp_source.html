229 static void CreateOrResizeBuffer(VkBuffer& buffer, VkDeviceMemory& buffer_memory, VkDeviceSize& p_buffer_size, size_t new_size, VkBufferUsageFlagBits usage)
230 {
231  ImGui_ImplVulkan_InitInfo* v = &g_VulkanInitInfo;
232  VkResult err;
233  if (buffer != VK_NULL_HANDLE)
234  vkDestroyBuffer(v->Device, buffer, v->Allocator);
235  if (buffer_memory != VK_NULL_HANDLE)
236  vkFreeMemory(v->Device, buffer_memory, v->Allocator);
237 
238  VkDeviceSize vertex_buffer_size_aligned = ((new_size - 1) / g_BufferMemoryAlignment + 1) * g_BufferMemoryAlignment;
239  VkBufferCreateInfo buffer_info = {};
240  buffer_info.sType = VK_STRUCTURE_TYPE_BUFFER_CREATE_INFO;
241  buffer_info.size = vertex_buffer_size_aligned;
242  buffer_info.usage = usage;
243  buffer_info.sharingMode = VK_SHARING_MODE_EXCLUSIVE;
244  err = vkCreateBuffer(v->Device, &buffer_info, v->Allocator, &buffer);
245  check_vk_result(err);
246 
247  VkMemoryRequirements req;
248  vkGetBufferMemoryRequirements(v->Device, buffer, &req);
249  g_BufferMemoryAlignment = (g_BufferMemoryAlignment > req.alignment) ? g_BufferMemoryAlignment : req.alignment;
250  VkMemoryAllocateInfo alloc_info = {};
251  alloc_info.sType = VK_STRUCTURE_TYPE_MEMORY_ALLOCATE_INFO;
252  alloc_info.allocationSize = req.size;
253  alloc_info.memoryTypeIndex = ImGui_ImplVulkan_MemoryType(VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT, req.memoryTypeBits);
254  err = vkAllocateMemory(v->Device, &alloc_info, v->Allocator, &buffer_memory);
255  check_vk_result(err);
256

262 // Render function
263 // (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
264 void ImGui_ImplVulkan_RenderDrawData(ImDrawData* draw_data, VkCommandBuffer command_buffer)
265 {
266  // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
267  int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
268  int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
269  if (fb_width <= 0 || fb_height <= 0 || draw_data->TotalVtxCount == 0)
270  return;
271 
272  ImGui_ImplVulkan_InitInfo* v = &g_VulkanInitInfo;
273 
274  // Allocate array to store enough vertex/index buffers
275  ImGui_ImplVulkanH_WindowRenderBuffers* wrb = &g_MainWindowRenderBuffers;
276  if (wrb->FrameRenderBuffers == NULL)
277  {
278  wrb->Index = 0;
279  wrb->Count = v->ImageCount;
280  wrb->FrameRenderBuffers = (ImGui_ImplVulkanH_FrameRenderBuffers*)IM_ALLOC(sizeof(ImGui_ImplVulkanH_FrameRenderBuffers) * wrb->Count);
281  memset(wrb->FrameRenderBuffers, 0, sizeof(ImGui_ImplVulkanH_FrameRenderBuffers) * wrb->Count);
282  }
283  IM_ASSERT(wrb->Count == v->ImageCount);
284  wrb->Index = (wrb->Index + 1) % wrb->Count;
285  ImGui_ImplVulkanH_FrameRenderBuffers* rb = &wrb->FrameRenderBuffers[wrb->Index];
286 
287  VkResult err;
288 
289  // Create or resize the vertex/index buffers
290  size_t vertex_size = draw_data->TotalVtxCount * sizeof(ImDrawVert);
291  size_t index_size = draw_data->TotalIdxCount * sizeof(ImDrawIdx);
292  if (rb->VertexBuffer == VK_NULL_HANDLE || rb->VertexBufferSize < vertex_size)
293  CreateOrResizeBuffer(rb->VertexBuffer, rb->VertexBufferMemory, rb->VertexBufferSize, vertex_size, VK_BUFFER_USAGE_VERTEX_BUFFER_BIT);
294  if (rb->IndexBuffer == VK_NULL_HANDLE || rb->IndexBufferSize < index_size)
295  CreateOrResizeBuffer(rb->IndexBuffer, rb->IndexBufferMemory, rb->IndexBufferSize, index_size, VK_BUFFER_USAGE_INDEX_BUFFER_BIT);
296 
297  // Upload vertex/index data into a single contiguous GPU buffer
298  {
299  ImDrawVert* vtx_dst = NULL;
300  ImDrawIdx* idx_dst = NULL;
301  err = vkMapMemory(v->Device, rb->VertexBufferMemory, 0, vertex_size, 0, (void**)(&vtx_dst));
302  check_vk_result(err);
303  err = vkMapMemory(v->Device, rb->IndexBufferMemory, 0, index_size, 0, (void**)(&idx_dst));
304  check_vk_result(err);
305  for (int n = 0; n < draw_data->CmdListsCount; n++)
306  {
307  const ImDrawList* cmd_list = draw_data->CmdLists[n];
308  memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
309  memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
310  vtx_dst += cmd_list->VtxBuffer.Size;
311  idx_dst += cmd_list->IdxBuffer.Size;
312  }
313  VkMappedMemoryRange range[2] = {};
314  range[0].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
315  range[0].memory = rb->VertexBufferMemory;
316  range[0].size = VK_WHOLE_SIZE;
317  range[1].sType = VK_STRUCTURE_TYPE_MAPPED_MEMORY_RANGE;
318  range[1].memory = rb->IndexBufferMemory;
319  range[1].size = VK_WHOLE_SIZE;
320  err = vkFlushMappedMemoryRanges(v->Device, 2, range);
321  check_vk_result(err);
322  vkUnmapMemory(v->Device, rb->VertexBufferMemory);
323  vkUnmapMemory(v->Device, rb->IndexBufferMemory);
324  }
325 
326  // Bind pipeline and descriptor sets:
327  {
328  vkCmdBindPipeline(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, g_Pipeline);
329  VkDescriptorSet desc_set[1] = { g_DescriptorSet };
330  vkCmdBindDescriptorSets(command_buffer, VK_PIPELINE_BIND_POINT_GRAPHICS, g_PipelineLayout, 0, 1, desc_set, 0, NULL);
331  }

333  // Bind Vertex And Index Buffer:
334  {
335  VkBuffer vertex_buffers[1] = { rb->VertexBuffer };
336  VkDeviceSize vertex_offset[1] = { 0 };
337  vkCmdBindVertexBuffers(command_buffer, 0, 1, vertex_buffers, vertex_offset);
338  vkCmdBindIndexBuffer(command_buffer, rb->IndexBuffer, 0, sizeof(ImDrawIdx) == 2 ? VK_INDEX_TYPE_UINT16 : VK_INDEX_TYPE_UINT32);
339  }
340 
341  // Setup viewport:
342  {
343  VkViewport viewport;
344  viewport.x = 0;
345  viewport.y = 0;
346  viewport.width = (float)fb_width;
347  viewport.height = (float)fb_height;
348  viewport.minDepth = 0.0f;
349  viewport.maxDepth = 1.0f;
350  vkCmdSetViewport(command_buffer, 0, 1, &viewport);
351  }
352 
353  // Setup scale and translation:
354  // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayMin is typically (0,0) for single viewport apps.
355  {
356  float scale[2];
357  scale[0] = 2.0f / draw_data->DisplaySize.x;
358  scale[1] = 2.0f / draw_data->DisplaySize.y;
359  float translate[2];
360  translate[0] = -1.0f - draw_data->DisplayPos.x * scale[0];
361  translate[1] = -1.0f - draw_data->DisplayPos.y * scale[1];
362  vkCmdPushConstants(command_buffer, g_PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 0, sizeof(float) * 2, scale);
363  vkCmdPushConstants(command_buffer, g_PipelineLayout, VK_SHADER_STAGE_VERTEX_BIT, sizeof(float) * 2, sizeof(float) * 2, translate);
364  }
365 
366  // Will project scissor/clipping rectangles into framebuffer space
367  ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
368  ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
369 
370  // Render command lists
371  int vtx_offset = 0;
372  int idx_offset = 0;
373  for (int n = 0; n < draw_data->CmdListsCount; n++)
374  {
375  const ImDrawList* cmd_list = draw_data->CmdLists[n];
376  for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
377  {
378  const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
379  if (pcmd->UserCallback)
380  {
381  // User callback (registered via ImDrawList::AddCallback)
382  pcmd->UserCallback(cmd_list, pcmd);
383  }
384  else
385  {
386  // Project scissor/clipping rectangles into framebuffer space
387  ImVec4 clip_rect;
388  clip_rect.x = (pcmd->ClipRect.x - clip_off.x) * clip_scale.x;
389  clip_rect.y = (pcmd->ClipRect.y - clip_off.y) * clip_scale.y;
390  clip_rect.z = (pcmd->ClipRect.z - clip_off.x) * clip_scale.x;
391  clip_rect.w = (pcmd->ClipRect.w - clip_off.y) * clip_scale.y;
392 
393  if (clip_rect.x < fb_width && clip_rect.y < fb_height && clip_rect.z >= 0.0f && clip_rect.w >= 0.0f)
394  {
395  // Negative offsets are illegal for vkCmdSetScissor
396  if (clip_rect.x < 0.0f)
397  clip_rect.x = 0.0f;
398  if (clip_rect.y < 0.0f)
399  clip_rect.y = 0.0f;
400 
401  // Apply scissor/clipping rectangle
402  VkRect2D scissor;
403  scissor.offset.x = (int32_t)(clip_rect.x);
404  scissor.offset.y = (int32_t)(clip_rect.y);
405  scissor.extent.width = (uint32_t)(clip_rect.z - clip_rect.x);
406  scissor.extent.height = (uint32_t)(clip_rect.w - clip_rect.y);
407  vkCmdSetScissor(command_buffer, 0, 1, &scissor);
408 
409  // Draw
410  vkCmdDrawIndexed(command_buffer, pcmd->ElemCount, 1, idx_offset, vtx_offset, 0);
411  }
412  }
413  idx_offset += pcmd->ElemCount;
414  }
415  vtx_offset += cmd_list->VtxBuffer.Size;
416  }

850 VkSurfaceFormatKHR ImGui_ImplVulkanH_SelectSurfaceFormat(VkPhysicalDevice physical_device, VkSurfaceKHR surface, const VkFormat* request_formats, int request_formats_count, VkColorSpaceKHR request_color_space)
851 {
852  IM_ASSERT(request_formats != NULL);
853  IM_ASSERT(request_formats_count > 0);
854 
855  // Per Spec Format and View Format are expected to be the same unless VK_IMAGE_CREATE_MUTABLE_BIT was set at image creation
856  // Assuming that the default behavior is without setting this bit, there is no need for separate Swapchain image and image view format
857  // Additionally several new color spaces were introduced with Vulkan Spec v1.0.40,
858  // hence we must make sure that a format with the mostly available color space, VK_COLOR_SPACE_SRGB_NONLINEAR_KHR, is found and used.
859  uint32_t avail_count;
860  vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, NULL);
861  ImVector<VkSurfaceFormatKHR> avail_format;
862  avail_format.resize((int)avail_count);
863  vkGetPhysicalDeviceSurfaceFormatsKHR(physical_device, surface, &avail_count, avail_format.Data);
864 
865  // First check if only one format, VK_FORMAT_UNDEFINED, is available, which would imply that any format is available
866  if (avail_count == 1)
867  {
868  if (avail_format[0].format == VK_FORMAT_UNDEFINED)
869  {
870  VkSurfaceFormatKHR ret;
871  ret.format = request_formats[0];
872  ret.colorSpace = request_color_space;
873  return ret;
874  }
875  else
876  {
877  // No point in searching another format
878  return avail_format[0];
879  }
880  }
881  else
882  {
883  // Request several formats, the first found will be used
884  for (int request_i = 0; request_i < request_formats_count; request_i++)
885  for (uint32_t avail_i = 0; avail_i < avail_count; avail_i++)
886  if (avail_format[avail_i].format == request_formats[request_i] && avail_format[avail_i].colorSpace == request_color_space)
887  return avail_format[avail_i];
888 
889  // If none of the requested image formats could be found, use the first available
890  return avail_format[0];
891  }
