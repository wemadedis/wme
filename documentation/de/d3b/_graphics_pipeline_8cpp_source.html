159 void GraphicsPipeline::CreatePipeline(ShaderInfo vertexShader, ShaderInfo fragmentShader, VkPrimitiveTopology primitiveTopology)
160 {
161  VkPipelineShaderStageCreateInfo vShaderStageInfo = GetPipelineStageInfo(VK_SHADER_STAGE_VERTEX_BIT, vertexShader);

178 
179 
180  VkGraphicsPipelineCreateInfo pipelineInfo = {};
181  pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
182  pipelineInfo.stageCount = 2;
183  pipelineInfo.pStages = shaderStages;
184  pipelineInfo.pVertexInputState = &vInputInfo;
185  pipelineInfo.pInputAssemblyState = &inputAssembly;
186  pipelineInfo.pViewportState = &viewportState;
187  pipelineInfo.pRasterizationState = &rasterizer;
188  pipelineInfo.pMultisampleState = &multisampleInfo;
189  pipelineInfo.pDepthStencilState = &depthStencil;
190  pipelineInfo.pColorBlendState = &colorBlend;
191  CreatePipelineLayout();

194  pipelineInfo.subpass = 0;
