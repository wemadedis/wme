152  }
153

156  createInfo.surface = _instance->GetSurface();

159  createInfo.imageFormat = optFormat.format;
160  createInfo.imageColorSpace = optFormat.colorSpace;
161  createInfo.imageExtent = optExtent;
162  createInfo.imageArrayLayers = 1;
163  createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT | VK_IMAGE_USAGE_STORAGE_BIT;

166  uint32_t queueFamilyIndices[] = {graphicsFamily.value(), presentFamily.value()};
167

170  createInfo.imageSharingMode = VK_SHARING_MODE_CONCURRENT;
171  createInfo.queueFamilyIndexCount = 2;
172  createInfo.pQueueFamilyIndices = queueFamilyIndices;
173  }
174  else

176  createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE;
177  createInfo.queueFamilyIndexCount = 0;
178  createInfo.pQueueFamilyIndices = nullptr;
179  }

182  createInfo.compositeAlpha = VK_COMPOSITE_ALPHA_OPAQUE_BIT_KHR;
